def \_percentile(vals: List\[float\], p: float) \-\> float:  
"""p in \[0,1\]. Simple linear interpolation percentile."""  
if not vals:  
return 0.0  
xs \= sorted(vals)  
if len(xs) \== 1:  
return xs\[0\]  
k \= (len(xs) \- 1\) \* p  
f \= math.floor(k)  
c \= math.ceil(k)  
if f \== c:  
return xs\[int(k)\]  
return xs\[f\] \+ (xs\[c\] \- xs\[f\]) \* (k \- f)  
def \_median(vals: List\[float\]) \-\> float:  
return \_percentile(vals, 0.5)  
def \_compute\_structural\_complexity(c: "Candidate") \-\> Dict\[str, float\]:

ps \= \[n / total for n in counts\]  
ent \= 0.0

ent \-= p \* math.log(p, 2\)

\# \------------------------------------------------------------  
\# Complexity Index (simple, transparent)  
\# \------------------------------------------------------------  
def complexity\_index(inst: VRPInstance) \-\> float:  
n \= len(inst.customers)  
\# capacity tightness (\>=1 is tight; \>1 infeasible)  
tight \= inst\_total\_demand(inst) / (inst.vehicles \* inst.capacity)  
\# demand entropy (normalized-ish)  
demands \= \[d for (\_, d) in inst.customers.values()\]  
total \= sum(demands) or 1  
ps \= \[d/total for d in demands\]  
ent \= \-sum(p\*math.log(p, 2\) for p in ps if p \> 0\)  
\# cluster overlap proxy: compare spread to inter-centroid separation  
pts \= \[inst.customers\[i\]\[0\] for i in range(1, n+1)\]  
cx \= sum(p\[0\] for p in pts)/n  
cy \= sum(p\[1\] for p in pts)/n  
spread \= sum(inst\_euclid(inst, (cx,cy), p) for p in pts)/n \# avg dist to centroid  
\# pick 3 rough centroids by x-quantiles as a cheap proxy  
xs \= sorted((p\[0\], idx+1) for idx, p in enumerate(pts))  
q1 \= xs\[int(0.25\*(n-1))\]\[1\]  
q2 \= xs\[int(0.50\*(n-1))\]\[1\]  
q3 \= xs\[int(0.75\*(n-1))\]\[1\]  
cpts \= \[inst.customers\[q\]\[0\] for q in (q1,q2,q3)\]  
sep \= 0.0  
pairs \= 0  
for i in range(len(cpts)):  
for j in range(i+1, len(cpts)):  
sep \+= inst\_euclid(inst, cpts\[i\], cpts\[j\])  
pairs \+= 1  
sep \= sep / pairs if pairs else 1.0  
overlap \= spread / (sep \+ 1e-9)  
\# index (bigger \= harder)  
return n \* (1.0 \+ overlap \+ tight \+ 0.25\*ent)

def compute\_compactness(depot, customers, routes):  
\# sum of avg distance of customers to route centroid  
total \= 0.0  
for r in routes:  
if not r:  
continue  
pts \= \[customers\[c\]\[0\] for c in r\]  
cx \= sum(p\[0\] for p in pts)/len(pts)  
cy \= sum(p\[1\] for p in pts)/len(pts)  
total \+= sum(euclid((cx,cy), p) for p in pts)/len(pts)  
return total  
def compute\_depot\_leg\_sum(depot, customers, routes):  
total \= 0.0  
for r in routes:  
if not r:  
continue  
total \+= euclid(depot, customers\[r\[0\]\]\[0\]) \+ euclid(customers\[r\[-1\]\]\[0\], depot)  
return total  
def compute\_spine\_score(depot, customers, routes, edge\_thresh=20.0):  
\# fraction of edges shorter than threshold (includes depot edges)  
short\_edges \= 0  
total\_edges \= 0  
for r in routes:  
prev \= depot  
for cid in r:  
total\_edges \+= 1  
if euclid(prev, customers\[cid\]\[0\]) \< edge\_thresh:  
short\_edges \+= 1  
prev \= customers\[cid\]\[0\]  
total\_edges \+= 1  
if euclid(prev, depot) \< edge\_thresh:  
short\_edges \+= 1  
return short\_edges/total\_edges if total\_edges else 0.0

def percentile(values, p):  
if not values:  
return 0.0  
xs \= sorted(values)  
if len(xs) \== 1:  
return xs\[0\]  
k \= (len(xs)-1) \* p  
f \= int(math.floor(k))  
c \= int(math.ceil(k))  
if f \== c:  
return xs\[f\]  
return xs\[f\] \+ (xs\[c\]-xs\[f\]) \* (k \- f)

\# Normalize by pool medians to keep stable across sizes.

def extract\_tight\_pairs\_from\_routes(depot, customers, routes, edge\_thresh=20.0):  
"""  
Returns list of (a,b) customer pairs that are adjacent in routes with short edge.  
Includes only customer-\>customer edges (not depot edges).  
"""  
pairs \= \[\]  
for r in routes:  
for i in range(len(r)-1):  
a, b \= r\[i\], r\[i+1\]  
if euclid(customers\[a\]\[0\], customers\[b\]\[0\]) \< edge\_thresh:  
pairs.append((a,b))  
return pairs  
def mine\_patterns\_from\_buffer(depot, customers, paradox\_buffer, edge\_thresh=20.0,  
top\_k\_pairs=30):

freq\[key\] \= freq.get(key, 0\) \+ 1

tight\_pairs \= sorted(freq.items(), key=lambda kv: kv\[1\], reverse=True)\[:top\_k\_pairs\]  
tight\_pairs \= \[pair for pair,count in tight\_pairs\]  
\# build clusters from tight pairs via union-find (quick & dirty)  
parent \= {}  
def find(x):  
parent.setdefault(x, x)  
if parent\[x\] \!= x:  
parent\[x\] \= find(parent\[x\])  
return parent\[x\]  
def union(a,b):  
ra, rb \= find(a), find(b)  
if ra \!= rb:  
parent\[rb\] \= ra  
for a,b2 in tight\_pairs:  
union(a,b2)

def apply\_cluster\_lock(assign, vehicles, cluster, rng):  
"""  
Force all customers in cluster to the same vehicle (chosen to minimize load impact).  
Returns a new assignment.  
"""  
a \= assign\[:\]  
\# choose target vehicle that currently has lowest load (approx by counts)  
cpv \= customers\_per\_vehicle\_from\_assign(a, vehicles)  
target \= min(range(vehicles), key=lambda v: cpv\[v\])  
for cid in cluster:  
a\[cid-1\] \= target  
\# lightly rebalance: random relocate a few non-cluster nodes if collapse is forming  
if is\_collapse\_trap(a, vehicles):  
\# move 3 random customers (not in cluster) to other vehicles  
cluster\_set \= set(cluster)  
non\_cluster \= \[i for i in range(1, len(a)+1) if i not in cluster\_set\]  
for \_ in range(min(3, len(non\_cluster))):  
cid \= rng.choice(non\_cluster)  
cur \= a\[cid-1\]  
other \= rng.choice(\[v for v in range(vehicles) if v \!= cur\])  
a\[cid-1\] \= other  
return a  
def apply\_spine\_split(assign, vehicles, spine, rng):  
"""  
Split a spine list across two vehicles to reduce overload while preserving adjacency.  
Strategy: split by x-coordinate median.  
"""  
a \= assign\[:\]  
\# pick two vehicles to split into  
v1, v2 \= rng.sample(range(vehicles), 2\)  
\# split spine into halves  
half \= len(spine)//2  
left \= spine\[:half\]  
right \= spine\[half:\]  
for cid in left:  
a\[cid-1\] \= v1  
for cid in right:  
a\[cid-1\] \= v2  
\# collapse guard  
if is\_collapse\_trap(a, vehicles):  
\# random scatter 2 nodes outside spine  
spine\_set \= set(spine)  
others \= \[i for i in range(1, len(a)+1) if i not in spine\_set\]  
for \_ in range(min(2, len(others))):  
cid \= rng.choice(others)  
a\[cid-1\] \= rng.randrange(vehicles)  
return a  
def apply\_capacity\_repair(assign, customers, vehicles, capacity, rng):  
"""  
Move smallest-demand customers off the most overloaded vehicle first.  
"""  
a \= assign\[:\]  
loads \= \[0\]\*vehicles  
for cid, v in enumerate(a, start=1):  
loads\[v\] \+= customers\[cid\]\[1\]  
\# find most overloaded vehicle  
v\_over \= max(range(vehicles), key=lambda v: loads\[v\] \- capacity)  
if loads\[v\_over\] \<= capacity:  
return a \# nothing to repair  
\# list customers on v\_over sorted by demand ascending  
cids \= \[cid for cid in range(1, len(a)+1) if a\[cid-1\] \== v\_over\]  
cids.sort(key=lambda cid: customers\[cid\]\[1\])  
for cid in cids:  
if loads\[v\_over\] \<= capacity:  
break  
\# move cid to least loaded vehicle  
v\_target \= min(range(vehicles), key=lambda v: loads\[v\])  
if v\_target \== v\_over:  
continue  
a\[cid-1\] \= v\_target  
loads\[v\_over\] \-= customers\[cid\]\[1\]  
loads\[v\_target\] \+= customers\[cid\]\[1\]  
return a  
def apply\_depot\_leg\_min(assign, depot, customers, vehicles, rng, k\_move=2):  
"""  
Move a few customers that are very far from their assigned vehicle centroid  
to the nearest centroid vehicle (reduces depot legs and route spans).  
"""  
a \= assign\[:\]  
\# compute centroid per vehicle  
centroids \= \[\]  
for v in range(vehicles):  
pts \= \[customers\[cid\]\[0\] for cid in range(1, len(a)+1) if a\[cid-1\] \== v\]  
if not pts:  
centroids.append(depot)  
else:  
cx \= sum(p\[0\] for p in pts)/len(pts)  
cy \= sum(p\[1\] for p in pts)/len(pts)  
centroids.append((cx,cy))  
\# find worst outliers: far from own centroid  
outliers \= \[\]  
for cid in range(1, len(a)+1):  
v \= a\[cid-1\]  
d \= euclid(customers\[cid\]\[0\], centroids\[v\])  
outliers.append((d, cid))  
outliers.sort(reverse=True)  
for \_, cid in outliers\[:k\_move\]:  
\# move to nearest centroid  
best\_v \= min(range(vehicles), key=lambda v: euclid(customers\[cid\]\[0\], centroids\[v\]))  
a\[cid-1\] \= best\_v  
\# collapse guard  
if is\_collapse\_trap(a, vehicles):  
for \_ in range(2):  
cid \= rng.randrange(1, len(a)+1)  
a\[cid-1\] \= rng.randrange(vehicles)  
return a

def validate\_pilot\_decision(

"""  
Strictly validate the pilot's iteration decision JSON.  
Raises PilotError on invalid.

if set(d.keys()) \!= REQUIRED\_DECISION\_KEYS:  
missing \= REQUIRED\_DECISION\_KEYS \- set(d.keys())  
extra \= set(d.keys()) \- REQUIRED\_DECISION\_KEYS  
raise PilotError(f"Pilot JSON keys mismatch. missing={sorted(missing)}  
extra={sorted(extra)}")  
\# keep\_ids  
if not isinstance(d\["keep\_ids"\], list) or not d\["keep\_ids"\]:  
raise PilotError("keep\_ids must be a non-empty list")  
bad \= \[x for x in d\["keep\_ids"\] if x not in candidate\_ids\]  
if bad:  
raise PilotError(f"keep\_ids contains unknown ids: {bad\[:5\]}")  
\# paradox\_add\_ids

raise PilotError("paradox\_add\_ids must be a list")  
if len(d\["paradox\_add\_ids"\]) \> paradox\_k:  
raise PilotError(f"paradox\_add\_ids must be \<= {paradox\_k}")  
badp \= \[x for x in d\["paradox\_add\_ids"\] if x not in candidate\_ids\]  
if badp:  
raise PilotError(f"paradox\_add\_ids contains unknown ids: {badp\[:5\]}")

total \= 0.0  
for k, v in mix.items():  
if not isinstance(v, (int, float)) or v \< 0:  
raise PilotError(f"operator\_mix\_next\[{k}\] must be a non-negative number")  
total \+= float(v)

raise PilotError(f"operator\_mix\_next must sum to 1.0 (got {total})")
